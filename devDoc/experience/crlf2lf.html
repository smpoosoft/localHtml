  • git 不同操作系统使用的换行符是不一样的。Unix/Linux使用的是LF，Mac后期也采用了LF，但Windows一直使用CRLF【回车(CR, ASCII 13, \r) 换行(LF, ASCII 10, \n)】作为换行符。而git入库的代码采用的是LF格式，它考虑到了跨平台协作的场景，提供了“换行符自动转换”的功能：如果在Windows下安装git，在拉取文件时，会自动将LF换行符替换为CRLF；在提交时，又会将CRLF转回LF。但是这个转换是有问题的：有时提交时，CRLF转回LF可能会不工作，尤其是文件中出现中文字符后有换行符时。 解决方案

禁用git的自动换行功能：

git config --global core.autocrlf false
git config --global core.filemode false
git config --global core.safecrlf true

自定义脚本

/* eslint-disable no-console */
const glob = require('glob-all');
const exec = require('child_process').exec;

// 解决 child_process 在 windows 下的 stdout 、stderr 乱码，替代  iconv-lite
/** TextDecoder 介绍：
 * 由于 child_process.execFileSync() 并不支持 Windows 所使用的 GBK 编码，
 * 通常会使用 iconv-lite 做转码方案。
 * 其实 Node.js 自带编码转换类 TextDecoder。
 * 值得注意的是，TextDecoder 需要 Node.js v8.3.0 及更新版本才能使用，
 * 如果 Node.js 版本过低，需要更新 nodeJs 或使用 iconv-lite
 *
 */
const decodeer = new TextDecoder('gbk');

const fmtStdStream = stdSome => decodeer.decode(Buffer.from(stdSome));

/** 异步执行命令行指令
 * @param cmdStr 要执行的指令
 */
const execCmd = cmdStr => {
	return new Promise((resovle, reject) => {
		exec(cmdStr, {
			maxBuffer: 1024 * 2000
		}, (err, stdout, stderr) => {
			if (err) {
				resovle(err.message);
			} else if (stderr.length) {
				resovle(stderr.toString());
			} else {
				resovle(stdout);
			}
		});
	});
};

/** 格式化 windows 路径符为 /
 *
 */
const formatPath = pathStr => (pathStr || '').replace(/\\\\/g, '/').replace(/\\/g, '/');

/** 执行 CRLF 到 LF 的批量转换
 * @param pjFolder 要批量转换的项目/工程的 根文件夹
 * @param doc2UnixPath dos2unix 可执行文件的路径，默认为 d:/Git/usr/bin/dos2unix
 */
const execDoc = async (pjFolder, doc2UnixPath = 'd:/Git/usr/bin/dos2unix') => {
	console.clear();	 // eslint-disable-line
	const _arr = glob.sync(`pjFolder/**/*`)?.filter(v => !v.includes('node_modules')) || [];
	let errMss = '';
	try {
		for (const v of _arr) {
			const fName = formatPath(v);
			const resExec = await execCmd(`${doc2UnixPath} ${fName}`);
			console.log(fmtStdStream(resExec));
			errMss = v;
		}
		console.log('Done!');	 // eslint-disable-line
	} catch (err) {
		console.log(errMss);
		console.error(fmtStdStream(err.message));
		process.exit(1);
	}
};
execDoc();
